#### 抽奖动画

```vue
<template>
 	<--以下样式仅占位-->
    <div class="store">
        <div class="goods">
            <div class="goods_item" v-for="(item, index) in 6" :key="index">
                <div class="icon"></div>
                <div class="goods_item_bottom">【蛋】</div>
                <img src="@/pages/celebration/assets/mk6_2.png"
                    v-show="selectFlag == index + 1 || selectArr.includes(index + 1)" alt="" class="selected" />
            </div>
        </div>
        <div class="awards">
            <div class="award_item" v-for="(item, index) in 5" :key="index">
                <div class="icon"></div>
                <div class="goods_item_bottom">+10</div>
                <--高亮效果边框-->
                <img class="selected" src="@/pages/celebration/assets/mk6_2.png"
                    v-show="selectFlag == index + 7 || selectArr.includes(index + 7)" alt="" />
            </div>
        </div>
        <div class="btns">
            <div class="left_btn" @click="draw(1)"></div>
            <div class="right_btn" @click="draw(10)"></div>
        </div>
    </div>
</template>

<script>
export default {
    data() {
        return {
            horseTimer: null, // 跑马灯定时器
            selectFlag: 1, // 当前抽奖奖池闪烁的item
            selectArr: [],
            drawState: false, // 抽奖状态 true-抽奖中 false-未抽奖
        }
    },
    destroyed() { },
    methods: {
        draw(times) {
            console.log(times, '抽奖次数');
            if (times == 1) {
                let num = Math.floor(Math.random() * 11) + 1
                this.horselight(80, num, 1, 1)
            } else {
                let arr = []
                for (let index = 0; index < 10; index++) {
                    arr.push(Math.floor(Math.random() * 11) + 1)
                }
                this.horselight(80, arr, 2, 1)
            }

        },
        // 抽奖跑马灯 
        // speed 定时器时间 
        // data 抽到的奖励index 
        // type 1 单抽 10 十连抽 
        // number 转的圈数 如一圈，一共11个格子 number就是11 两圈就是22
        horselight(speed, data, type, number) {
            console.log(data, '抽奖数据');
            clearInterval(this.horseTimer)
            let nums = 0
            this.selectArr = []
            // 产品特殊需求 
            // 十抽 固定第一个盒子开始动画
            // 单抽 如果当前用户没有抽奖 那么从一个盒子开始；如果抽奖过了，从抽到盒子开始动画（代码方面不用做特殊处理）
            if (type == 2) {
                this.selectFlag = 1
            }
            //---------------------------
            this.horseTimer = setInterval(() => {
                if (type == 1) {
                    // 抽取一次
                    if ((nums / 11) >= number && this.selectFlag == data) {
                        // 转完
                        clearInterval(this.horseTimer)
                        this.drawState = false
                    } else {
                        // 没转完
                        nums++
                        if (this.selectFlag == 11) {
                            this.selectFlag = 1
                        } else {
                            this.selectFlag++
                        }
                    }
                } else {
                    // 抽取十次
                    if ((nums / 11) >= number) {
                        // console.log(this.selectFlag, 3333);
                        // 转完
                        if (data.includes(this.selectFlag)) {
                            console.log(this.selectFlag, 444);
                            let i = this.selectFlag
                            this.selectArr.push(i)
                        }
                        // 转到最后一个奖励就停止 判断获奖index中最大的值 转到该index意味着动画结束
                        if (this.selectFlag >= Math.max(...data)) {
                            clearInterval(this.horseTimer)
                            this.drawState = false
                        } else {
                            this.selectFlag++
                        }
                    } else {
                        // 没转完
                        nums++
                        if (this.selectFlag == 11) {
                            this.selectFlag = 1
                        } else {
                            this.selectFlag++
                        }
                    }
                }

            }, speed)
        }
    }
}
</script>
```
