4. ```js
   clipboard-polyfill
   <script type="module">
       // 参考文档 https://www.ruanyifeng.com/blog/2021/01/clipboard-api.html
       // 目前结论 只有在https协议下才能使用clipboard API或者本地localhost环境下
       // 并且需要剪贴板权限
       // 没有权限得报错信息为Read is not supported in your browser
       // 非https环境下也是同样的报错，在控制台直接访问navigator.clipboard是undefined
       new window.VConsole({ defaultPlugins: ['system', 'network'] });
       import * as clipboard from 'https://cdnjs.cloudflare.com/ajax/libs/clipboard-polyfill/4.1.0/es6/clipboard-polyfill.es6.min.js';
       const input = document.querySelector("input");
   
       function copy () {
         clipboard.writeText("This text is plain.").then(
           () => { console.log("success!"); },
           (error) => { console.log(error); }
         );
       }
   
       function past () {
         clipboard.readText().then(
           (text) => { input.value = text; },
           (error) => { console.log(error); }
         );
       }
   
       Copy.addEventListener("click", copy);
       Past.addEventListener("click", past);
   </script>
   ```

5. postcssPresetEnv：用来提高css兼容，转换新特性和添加浏览器前缀；

   ```js
   // npm i postcss-preset-env --save-dev
   // postcss-preset-env 插件有许多配置项，以下是一些主要的配置项：
   // stage：决定你需要哪个阶段的 CSS 特性。默认值为 2。这个选项的值可以是 0 到 4 的整数，0 表示你需要所有阶段的特性，包括还在草案阶段的特性，4 表示你只需要稳定的特性。
   // browsers：决定你需要支持哪些浏览器。这个选项的值可以是一个浏览器查询字符串，例如 'last 2 versions' 表示你需要支持所有浏览器的最后两个版本。
   // autoprefixer：决定是否使用 Autoprefixer 来添加浏览器前缀。默认值为 true。
   // preserve：决定是否保留原始的 CSS。默认值为 false。
   // importFrom：指定一个或多个文件，插件会从这些文件中导入自定义属性、自定义媒体查询和环境变量。
   // exportTo：指定一个或多个文件，插件会将自定义属性、自定义媒体查询和环境变量导出到这些文件。
   // features：决定你需要哪些 CSS 特性。这个选项的值可以是一个对象，对象的键是特性的名称，值是一个布尔值或者一个配置对象。
   'postcss-preset-env': {
     stage: 0,
     browsers: 'last 18 versions'
   }
   ```
