// urls 请求列表 max 最大请求书
function request(urls, max) {
  return new Promise((resolve, reject) => {
    let index = 0;
    let result = []; // 结果数据
    let total = 0; // 完成请求的次数
    // 请求函数 
    function _request() {
      let i = index; // 用于保证结果数组的顺序 第一个请求的结果也必须在结果数组的第一个
      index++; // 记录是第几次请求 用于获取请求信息
      axios({
        method: "POST",
        url: urls[i], // 为什么不用index 因为index++ 需要用i获得正确的请求信息
        data: {
          name: "cyg",
          password: "cyg20240207!",
        },
      })
        .then((res) => {
          // 为什么不用index 因为index++ 需要用i保存正确的结果顺序 错误结果同理
          result[i] = res; // 记录成功结果
        })
        .catch((err) => {
          result[i] = err; // 记录失败结果
        })
        .finally(() => {
          total++; // 完成请求 次数+1
          // 完成一次请求之后 判断请求次数 未超过开始就递归
          // 为什么不用total字段判断 total字段只有完成后才会自增 请求是异步
          // 假如urls长度是20 已经发出去了20个请求 index等于20 但是只有16个返回结果 total等于16
          // 再次调用_request方法 index等于20 但是urls并没有第20项 会报错进入catch 保存结果
          // result就会有24项 20项正常请求结果 4项没找到请求信息的报错信息
          if (index < urls.length) {
            _request();
          } else if (total == urls.length) { // 全部请求完成 触发resolve回调 返回结果
            // 该判断用于是否返回结果 返回结果的前提是需要完成所有结果 所以这个判断在finally回调中执行
            // 为什么不使用index字段判断 返回结果的前提是需要完成所有结果 而index字段在请求发出去的时候已经自增了
            // 如果发出去的请求次数已经达到urls的长度 也就说请求已经发送完毕了 但是结果并没有返回 因为请求是异步的
            // 比如urls长度20 已经发出去了20个请求 也就是index等于20 但是返回来的只有16个结果 并没有完成所有请求
            resolve(result);
          }
        });
    }
    // 循环发送请求 
    for (let index = 0; index < max; index++) {
      _request();
    }
  });
}
