代码配方

1. 读取 ZIP 结构

const { data } = await axios.get(zipUrl, {
  responseType: 'arraybuffer'
});

const zip = await JSZip.loadAsync(data);
console.log(Object.keys(zip.files));

2. 读取 JSON 文件

const jsonFile = zip.file('data/info.json');
if (!jsonFile) throw new Error('没有找到 JSON');

const jsonText = await jsonFile.async('string');
const jsonData = JSON.parse(jsonText);

3. 读取图片并转成 data URL

const imageFile = zip.file('images/demo.svg');
const base64 = await imageFile.async('base64');
const imageUrl = `data:image/svg+xml;base64,${base64}`;

4. 读取 MP4 或其他媒体文件

const videoFile = zip.file('video/demo.mp4');
const videoBlob = await videoFile.async('blob');
const videoUrl = URL.createObjectURL(videoBlob);

video.src = videoUrl;
video.load();
